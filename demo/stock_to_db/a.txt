
select BrandName,PartType,t5.PositionName,Qty,PartNo,SubBrandName
from www_gzqp_dbqp_com.product t1
left join www_gzqp_dbqp_com.brand t2 on t1.BrandId = t2.BrandId
left join www_gzqp_dbqp_com.subbrand t3 on t3.BrandId = t2.BrandId
left join www_gzqp_dbqp_com.productfitment t4 on t4.ProductId = t1.ProductId
left join www_gzqp_dbqp_com.positioninfo t5 on t5.PositionInfoId = t4.PositionId
join www_gzqp_dbqp_com.systemproduct t6 on t6.ReferenceCode = t1.ReferenceCode
where PositionName is not null and PositionName != 'N/A' and PartType is not null and PartType != ''
and PartType = 'Exterior Door Handle' and PositionName = 'Front' and Qty = 1
group by BrandName,PartType,t5.PositionName,Qty,PartNo,SubBrandName
;
